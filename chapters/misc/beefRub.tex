\section{Beef Rub}
\label{beefRub}
\setcounter{secnumdepth}{0}
Time: 1 minute
Serves: 1 recipe of beef rub

\begin{multicols}{2}
\subsection*{Ingredients}
\begin{itemize}
    \item \( \frac{1}{4} \) cup cracked black pepper, 16 mesh
    \item \( \frac{1}{4} \) cup kosher salt
\end{itemize}

\subsection*{Hardware}
\begin{itemize}
    \item 1 Large shaker
\end{itemize}
\clearpage

\subsection*{Instructions}
\begin{enumerate}
    \item Combine both ingredients in the shaker.
\end{enumerate}

\subsection*{Notes}
\begin{itemize}
    \item This is Aaron Franklin's beef rub, as seen in Franklin Barbecue: A Meat-Smoking Manifesto.
\end{itemize}
\end{multicols}
\clearpage
